    \begin{equation*} A' = X \setminus A = \{ x : x \in X \text{ y } x \notin A \}\text{.} \end{equation*}

    \begin{equation*} a \wedge ( b \vee c ) = (a \wedge b ) \vee ( a \wedge c ) \end{equation*}
