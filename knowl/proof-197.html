\begin{equation*} \sigma_i(a \pm b) = \sigma_i(a) \pm \sigma_i(b) = a \pm b \end{equation*}

\begin{equation*} \sigma_i(a b) = \sigma_i(a) \sigma_i(b) = a b\text{.} \end{equation*}
